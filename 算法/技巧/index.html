    前缀和¶

    计算一维

    • 一维情况
    \[ S[i] = S[i-1] + A[i] \]

    ​ 其中\(A[i]+\cdots +A[j] = S[j] - S[i-1]\)

    • 二维情况

    $$ S[i][j] = S[i-1][j] + S[i][j-1] - S[i-1][j-1]+ A[i][j] $$ 其中\(x_1,y_1\)到\(x_2,y_2\)的子矩阵的和为\(S[x_2][y_2] - S[x_1-1][y_2] - S[x_2][y_1-1] + S[x_1-1][y_1-1]\)

    差分数组¶

    对于数组\(a\)，其差分数组为

    \[ d[i] = a[i] - a[i-1] \]
    • 从左到右累加\(d\)中的元素，可以得到数组\(a\)
    • 可以将区间操作转换为单点操作
    • 区间操作：把\(a\)的子数组\(a[i],a[i+1],\cdots,a[j]\)都加上\(x\)
    • 单点操作：把\(d[i]\)增加\(x\)，把\(d[j+1]\)减少\(x\)，如果\(j+1==n\)只需把\(d[i]\)加\(x\)

    // 你有一个长为 n 的数组 a，一开始所有元素均为 0。
// 给定一些区间操作，其中 queries[i] = [left, right, x]，
// 你需要把子数组 a[left], a[left+1], ... a[right] 都加上 x。
// 返回所有操作执行完后的数组 a。
vector<int> solve(int n, vector<vector<int>> queries) {
    vector<int> diff(n); // 差分数组
    for (auto &q: queries) {
        int left = q[0], right = q[1], x = q[2];
        diff[left] += x;
        if (right + 1 < n) {
            diff[right + 1] -= x;
        }
    }
    for (int i = 1; i < n; i++) {
        diff[i] += diff[i - 1]; // 直接在差分数组上复原数组 a
    }
    return diff;
}

    差分矩阵¶

    与差分数组类似，对差分矩阵b[m][n]求前缀和可以得到原始矩阵a[m][n]。由原始矩阵a[m][n]求其差分矩阵b[m][n]则是一件比较困难的事情。

    类比差分数组，差分矩阵的出现是为了方便使用\(O(1)\)的时间对子矩阵的元素进行统一的增减，当对原矩阵a[x1][y1]到a[x2][y2]之间的子矩阵的元素都加上c时，等价于对差分矩阵进行如下操作

    $$ \begin{gather} b[x_1,y_1] += c \ b[x_2+1,y_1] -= c \ b[x_1,y_2+1] -= c \ b[x_2+1,y_2+1] += c \ \end{gather} $$ 于是，从矩阵a计算它的差分矩阵可以这样构造

    • 假设此时对应的矩阵a也是0矩阵，初始化矩阵b为元素全为0的矩阵
    • 对矩阵a的每一个位置填上原来的数字，转换为对应的对矩阵b的操作

    class Solution {
public:
    vector<vector<int>> dp;

    void insert(int x1, int y1, int x2, int y2) {
        dp[x1][y1] += 1;
        dp[x2 + 1][y1] -= 1;
        dp[x1][y2 + 1] -= 1;
        dp[x2 + 1][y2 + 1] += 1;
    }

    vector<vector<int>> rangeAddQueries(int n, vector<vector<int>> &queries) {
        this->dp = vector<vector<int>>(n + 1, vector<int>(n + 1, 0));
        for (auto q: queries) insert(q[0], q[1], q[2], q[3]);
        vector<vector<int>> answer(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0 && j == 0) answer[i][j] = dp[i][j];
                else if (i == 0) answer[i][j] = answer[i][j - 1] + dp[i][j];
                else if (j == 0) answer[i][j] = answer[i - 1][j] + dp[i][j];
                else answer[i][j] = answer[i - 1][j] + answer[i][j - 1] - answer[i - 1][j - 1] + dp[i][j];
            }
        }
        return answer;
    }
};

    连续数组（哈希表+前缀和）¶

    给定一个二进制数组 nums , 找到含有相同数量的 0 和 1的最长连续子数组，并返回该子数组的长度。

    思路

    将数组中的0看做-1，就变成了统计最长的和为0的连续子数组。连续子数组的和为0，意味着连续子数组第一个元素和最后一个元素的前缀和是一样的。那么我们只需要记录每一个不同值的前缀和第一次出现的位置，就代表着对应的一个连续子数组和为0的一个起点。

    int findMaxLength(vector<int>& nums) {
    int n = nums.size();
    unordered_map<int,int> mp;
    mp[0] = -1;
    int counter = 0;
    int res = 0;
    for(int i=0;i<n;i++){
        if(nums[i]==1) counter++;
        else counter--;
        if(mp.find(counter)==mp.end()) mp[counter] = i;
        else res = max(res,i-mp[counter]);
    }
    return res;
}

    bool canReach(string s, int minJump, int maxJump) {
    int n = s.size();
    vector<bool> dp(n,false);
    vector<int> pre(n,0);
    dp[0] = true;
    int total;
    for(int i=0;i<minJump;i++) pre[i] = 1;
    for(int i=minJump;i<n;i++){
        int left = max(i-maxJump,0);
        int right = min(i-minJump,n-1);
        if(s[i]=='0'){
            if(left==0) total = pre[right];
            else total = pre[right] - pre[left-1];
            dp[i] = (total>0);
        }
        pre[i] = pre[i-1]+dp[i];
    } 
    return dp[n-1];
}

    双指针¶

    基本思想¶

    双指针是指在遍历对象时，使用两个或多个指针进行遍历及相应的操作。大多用于数组操作，这利用了数组连序性的特点。双指针常用来降低算法的时间复杂度，因为使用两个指针可以避免多层循环。

    要素¶

    • 指针的起始位置的选取
    • 指针的移动方向
    • 指针的移动速度

    类型¶

    • 对撞指针：左右两个指针，向中间靠拢
    • 快慢指针：左右两个指针，一块一慢
    • 滑动窗口：左右两个指针组成一个"窗口",右指针不断扩张，左指针按条件收缩
    • 归并排序

    例题¶

    三数之和¶

    给你一个整数数组 nums ，判断是否存在三元组 [nums[i], nums[j], nums[k]] 满足 i != j、i != k且 j != k ，同时还满足 nums[i] + nums[j] + nums[k] == 0 。请

    你返回所有和为 0 且不重复的三元组。

    注意：答案中不可以包含重复的三元组。

    思路¶

    使用对撞指针，首先对数组进行排序，遍历数组中的每一个元素，设置对撞指针l和r，\(sum=n[i]+n[l]+[r]\)，当\(sum>0\)，\(r--\)，反之\(l++\)，等于零则添加一种情况。此外，遍历过程中需要两次驱去重，第一次是保证n[i]和上一次的不重复，第二次是保证n[l]和上一次的不重复。

        while (l < r) {

            if (sum == 0) {
                ans.push_back({nums[i], nums[l], nums[r]});
                int temp = nums[l];
                while (l < r && temp == nums[l]) l++;

            } else if (sum > 0) r--;
            else l++;
        }
    }
    return ans;

    循环¶

    问题¶

    存在一个不含 0 的 环形 数组 nums ，每个 nums[i] 都表示位于下标 i 的角色应该向前或向后移动的下标个数：

    • 如果 nums[i] 是正数，向前（下标递增方向）移动 |nums[i]| 步
    • 如果 nums[i] 是负数，向后（下标递减方向）移动 |nums[i]| 步

    因为数组是 环形 的，所以可以假设从最后一个元素向前移动一步会到达第一个元素，而第一个元素向后移动一步会到达最后一个元素。

    • 遵循上述移动规则将导致一组重复下标序列 seq[0] -> seq[1] -> ... -> seq[k - 1] -> seq[0] -> ...
    • 所有 nums[seq[j]] 应当不是 全正 就是 全负
    • k > 1

    bool circularArrayLoop(vector<int>& nums) {
      int n = nums.size();
      auto next = [&](int cur) {
          return ((cur + nums[cur]) % n + n) % n; // 保证返回值在 [0,n) 中
      };

      for (int i = 0; i < n; i++) {
          if (!nums[i]) {
              continue;
          }
          int slow = i, fast = next(i);
          // 判断非零且方向相同
          while (nums[slow] * nums[fast] > 0 && nums[slow] * nums[next(fast)] > 0) {
              if (slow == fast) {
                  if (slow != next(slow)) {
                      return true;
                  } else {
                      break;
                  }
              }
              slow = next(slow);
              fast = next(next(fast));
          }
          int add = i;
          while (nums[add] * nums[next(add)] > 0) {
              int tmp = add;
              add = next(add);
              nums[tmp] = 0;
          }
      }
      return false;
}

    class Solution {
public:
    int help(vector<int>& nums, int firstLen, int secondLen) {
        int suml = accumulate(nums.begin(), nums.begin() + firstLen, 0);
        int maxSumL = suml;
        int sumr = accumulate(nums.begin() + firstLen, nums.begin() + firstLen + secondLen, 0);

            sumr += nums[i] - nums[i - secondLen];
            res = max(res, maxSumL + sumr);
        }
        return res;
    }

    int countNodes(TreeNode* root) {
    if (root == nullptr) {
        return 0;
    }
    int level = 0;
    TreeNode* node = root;
    while (node->left != nullptr) {
        level++;
        node = node->left;
    }
    int low = 1 << level, high = (1 << (level + 1)) - 1;
    while (low < high) {
        int mid = (high - low + 1) / 2 + low;
        if (exists(root, level, mid)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }
    return low;
}

bool exists(TreeNode* root, int level, int k) {

            node = node->left;
        } else {
            node = node->right;
        }

    class CustomStack {
public:
    vector<int> st,add;
    int top;
    CustomStack(int maxSize) {
        st.resize(maxSize,0);
        add.resize(maxSize,0);
        top = -1;
    }

    void push(int x) {
        if(top==(st.size()-1)) return;
        st[++top] = x;
    }

    int pop() {
        if(top==-1) return -1;
        int res = st[top] + add[top];
        if(top!=0){
            add[top-1] += add[top];
        }
        add[top] = 0;
        top--;
        return res;
    }

    void increment(int k, int val) {
        int j = min(k-1,top);
        if(j>=0) add[j] += val;
    }
};

    long long maximumSumOfHeights(vector<int>& maxHeights) {
    int n = maxHeights.size();
    vector<long long> dp1(n+1,0); // ->  
    vector<long long> dp2(n+1,0); // <-

        while(st1.size()>1&&maxHeights[i]<=maxHeights[st1.top()]){
            int j = st1.top();
            st1.pop();
            sum1 -= (long long) maxHeights[j] * (j-st1.top());

        st2.push(n-i-1);
        dp2[n-1-i] = sum2;
    }

    long long res = 0;
    for(int i=0;i<n;i++){
        res = max(res,dp1[i]+dp2[i+1]);
    }

    long long subArrayRanges(vector<int>& nums) {
    int n = nums.size();
    stack<int> stmax,stmin;
    stmax.push(-1),stmin.push(-1);
    vector<int> lmin(n,-1),rmin(n,n);

        while(stmax.size()>1&&nums[i]>=nums[stmax.top()]){
            rmax[stmax.top()] = i; //闭
            stmax.pop();
        }
        lmax[i] = stmax.top();//开
        stmax.push(i);

        - (long long)nums[i] * (long long)(rmin[i]-i)*(long long)(i-lmin[i]);
    }

    int maxSumMinProduct(vector<int>& nums) {
    int n = nums.size();
    stack<int> st;

    • 使用单调栈，栈底到栈顶单调递减，对于最长的和大于0的区间，其左端点一定在里面
    • 单调栈建立好之后，倒序遍历数组即可。

    int longestWPI(vector<int>& hours) {
    int n = hours.size();
    stack<int> st;
    vector<int> prefix(n+1,0);
    st.push(-1);
    for(int i=0;i<n;i++){
        if(hours[i]>8) prefix[i+1] = prefix[i]+1;
        else prefix[i+1] = prefix[i]-1;
        if(prefix[i+1]<prefix[st.top()+1]) st.push(i);
    }
    int res = 0;
    for(int i=n-1;i>=0;i--){
        while(!st.empty()&&prefix[i+1]>prefix[st.top()+1]){
            res = max(res,i-st.top());
            st.pop();
        }
    }
    return res;
}
